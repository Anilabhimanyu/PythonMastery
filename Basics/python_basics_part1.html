    print("Small")

For loop & range()

for i in range(5):   # 0..4
    print(i)

for i, val in enumerate(['a','b']):
    print(i, val)

While & control keywords

i = 0
while i < 3:
    i += 1
    if i == 2:
        continue
    if i == 3:
        break
else:
    # else runs if loop completes without break
    pass
Loop flow: start -> condition? -> true -> body -> update -> back to condition -> false -> exit
9 â–¸ Functions

Define with def. Functions are first-class objects.

def add(a, b=0):
    return a + b

print(add(2,3))    # 5

# variable arguments
def f(*args, **kwargs):
    print(args, kwargs)

f(1,2,x=3)

Lambda, map, filter, reduce

# lambda
square = lambda x: x*x
print(square(3))   # 9

# map & filter
nums = [1,2,3,4]
print(list(map(lambda x: x*2, nums)))
print(list(filter(lambda x: x%2==0, nums)))

# reduce (from functools)
from functools import reduce
print(reduce(lambda a,b: a+b, nums))  # 10

12 â–¸ Memory & Garbage Collection

Reference counting is primary in CPython; objects are freed when refcount drops to zero. The gc module handles cyclic garbage.

Memory (simplified): name -> object a -> [1,2,3] (list object on heap) b -> same object (a and b reference same id) 
import gc
print(gc.isenabled())   # True usually

# shallow vs deep copy
import copy
lst = [[1],[2]]
s = copy.copy(lst)      # shallow copy (inner lists shared)
d = copy.deepcopy(lst)  # deep copy (full clone)

Use immutable types when you want safe hashable objects (e.g., tuple as dict key).
